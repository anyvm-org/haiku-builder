

NEW_PASS="anyvm.org"

passwd <<EOF
$NEW_PASS
$NEW_PASS
EOF


echo "PermitRootLogin yes" >>/boot/system/settings/ssh/sshd_config

echo "PermitEmptyPasswords yes" >>/boot/system/settings/ssh/sshd_config
echo "PasswordAuthentication yes" >>/boot/system/settings/ssh/sshd_config

echo 'AcceptEnv   *'  >> /boot/system/settings/ssh/sshd_config



ssh-keygen -t rsa -f /boot/home/config/settings/ssh/id_rsa -q -N ""

echo "StrictHostKeyChecking=no" >>/boot/home/config/settings/ssh/config



rm -rf ~/.ssh
ln -s  /boot/home/config/settings/ssh  ~/.ssh

chmod 600 ~/.ssh

kill $(ps | grep sshd | grep -v grep | awk '{print $3}')
sleep 3;

/bin/sshd
